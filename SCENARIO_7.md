// 사용자별 상태 관리
interface UserState {
  entryOrder: number;
  heartsBalance: number;
  heartAllowance: number;
  currentLevel: number;
  currentFarm: number;
}

// 순차 입장 시뮬레이션
function simulateSequentialEntry(totalUsers: number) {
  const users: Map<number, UserState> = new Map();
  
  for (let entryOrder = 1; entryOrder <= totalUsers; entryOrder++) {
    // 신규 사용자 초기화
    users.set(entryOrder, {
      entryOrder,
      heartsBalance: 5, // 환영 하트
      heartAllowance: 0,
      currentLevel: 0,
      currentFarm: 1,
    });
    
    // 모든 기존 사용자의 허용치 +1
    for (const [userId, user] of users.entries()) {
      if (userId < entryOrder) {
        user.heartAllowance += 1;
      }
    }
    
    // 모든 사용자의 레벨업 시도
    for (const [userId, user] of users.entries()) {
      attemptLevelUp(user);
    }
  }
  
  return users;
}
